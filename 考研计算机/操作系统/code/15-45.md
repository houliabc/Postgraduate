```c
信号量;
mutex_A=1;用来互斥访问信箱A
mutex_B=1;用来互斥访问信箱B
A_Max=M;用来表示A邮箱的最大数量
B_Max=N;用来表示B邮箱的最大数量
A_Now=x;用来表示A邮箱当前的数量
B_Now=y;用来表示B邮箱当前的数量

A() {   // A人
    while (TRUE) {
        从A的邮箱中取出一个邮件;
        V(A_Max);
        V(A_Now);
        回答问题并提出一个新问题;
        P(mutex_B);
        P(B_Max);
        P(B_Now);
        将新邮件放入B的邮箱;
        V(mutex_B);
    }
}

B() {   // B人
    while (TRUE) {
        从B的邮箱中取出一个邮件;
        V(B_Max);
        V(B_Now);
        回答问题并提出一个新问题;
        P(mutex_A);
        P(A_Max);
        P(A_Now);
        将新邮件放入A的邮箱;
        V(mutex_A);
    }
}
